    public int[] decode(int[] encoded, int first) {
        int len = encoded.length;
        int[] result = new int[len + 1];
        result[0] = first;
        for (int i = 1; i < len + 1; i++) {
            result[i] = encoded[i -1] ^ result[i - 1];
        }

        return result;
    }
